# Database Integration

Patterns for using atom with SQL databases.

## Overview

Atom provides two paths for database integration:

1. **Scanner** - Direct row scanning into atoms (recommended)
2. **Manual** - Query into structs, then atomize

## Scanner Pattern

### Basic Query

```go
type User struct {
    ID    int64  `db:"id"`
    Name  string `db:"name"`
    Email string `db:"email"`
}

func GetUsers(db *sql.DB) ([]*atom.Atom, error) {
    atomizer, _ := atom.Use[User]()
    scanner, _ := atom.ScannerFor(atomizer.Spec())

    rows, err := db.Query("SELECT id, name, email FROM users")
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    return scanner.ScanAll(rows, rows.Next)
}
```

### With sqlx

```go
import "github.com/jmoiron/sqlx"

func GetUser(db *sqlx.DB, id int64) (*atom.Atom, error) {
    atomizer, _ := atom.Use[User]()
    scanner, _ := atom.ScannerFor(atomizer.Spec())

    rows, err := db.Queryx("SELECT id, name, email FROM users WHERE id = ?", id)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    if rows.Next() {
        return scanner.Scan(rows)
    }
    return nil, sql.ErrNoRows
}
```

### Batch Processing

```go
func ProcessUsers(db *sql.DB, batchSize int) error {
    atomizer, _ := atom.Use[User]()
    scanner, _ := atom.ScannerFor(atomizer.Spec())

    offset := 0
    for {
        rows, err := db.Query(
            "SELECT id, name, email FROM users LIMIT ? OFFSET ?",
            batchSize, offset,
        )
        if err != nil {
            return err
        }

        atoms, err := scanner.ScanAll(rows, rows.Next)
        rows.Close()
        if err != nil {
            return err
        }

        if len(atoms) == 0 {
            break
        }

        for _, a := range atoms {
            // Process each atom
            processUser(a)
        }

        offset += batchSize
    }
    return nil
}
```

## Repository Pattern

```go
type UserRepository struct {
    db       *sql.DB
    atomizer *atom.Atomizer[User]
    scanner  *atom.Scanner
}

func NewUserRepository(db *sql.DB) (*UserRepository, error) {
    atomizer, err := atom.Use[User]()
    if err != nil {
        return nil, err
    }

    scanner, ok := atom.ScannerFor(atomizer.Spec())
    if !ok {
        return nil, fmt.Errorf("scanner not available for User")
    }

    return &UserRepository{
        db:       db,
        atomizer: atomizer,
        scanner:  scanner,
    }, nil
}

func (r *UserRepository) FindByID(id int64) (*User, error) {
    rows, err := r.db.Query("SELECT id, name, email FROM users WHERE id = ?", id)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    if !rows.Next() {
        return nil, sql.ErrNoRows
    }

    a, err := r.scanner.Scan(rows)
    if err != nil {
        return nil, err
    }

    return r.atomizer.Deatomize(a)
}

func (r *UserRepository) FindAll() ([]*User, error) {
    rows, err := r.db.Query("SELECT id, name, email FROM users")
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    atoms, err := r.scanner.ScanAll(rows, rows.Next)
    if err != nil {
        return nil, err
    }

    users := make([]*User, len(atoms))
    for i, a := range atoms {
        user, err := r.atomizer.Deatomize(a)
        if err != nil {
            return nil, err
        }
        users[i] = user
    }
    return users, nil
}
```

## Insert/Update with Atoms

Scanner handles reads. For writes, use atoms directly:

### Insert

```go
func InsertUser(db *sql.DB, a *atom.Atom) error {
    _, err := db.Exec(
        "INSERT INTO users (id, name, email) VALUES (?, ?, ?)",
        a.Ints["ID"],
        a.Strings["Name"],
        a.Strings["Email"],
    )
    return err
}
```

### Bulk Insert

```go
func InsertUsers(db *sql.DB, atoms []*atom.Atom) error {
    tx, err := db.Begin()
    if err != nil {
        return err
    }
    defer tx.Rollback()

    stmt, err := tx.Prepare("INSERT INTO users (id, name, email) VALUES (?, ?, ?)")
    if err != nil {
        return err
    }
    defer stmt.Close()

    for _, a := range atoms {
        _, err := stmt.Exec(
            a.Ints["ID"],
            a.Strings["Name"],
            a.Strings["Email"],
        )
        if err != nil {
            return err
        }
    }

    return tx.Commit()
}
```

### Update

```go
func UpdateUser(db *sql.DB, a *atom.Atom) error {
    _, err := db.Exec(
        "UPDATE users SET name = ?, email = ? WHERE id = ?",
        a.Strings["Name"],
        a.Strings["Email"],
        a.Ints["ID"],
    )
    return err
}
```

## Nullable Columns

### Reading Nulls

```go
type User struct {
    ID       int64   `db:"id"`
    Name     string  `db:"name"`
    Nickname *string `db:"nickname"` // Nullable
}

func GetUser(db *sql.DB, id int64) (*atom.Atom, error) {
    scanner, _ := atom.ScannerFor(atomizer.Spec())

    rows, _ := db.Query("SELECT id, name, nickname FROM users WHERE id = ?", id)
    defer rows.Close()

    if rows.Next() {
        a, _ := scanner.Scan(rows)

        // Check nullable field
        if a.StringPtrs["Nickname"] != nil {
            fmt.Println("Nickname:", *a.StringPtrs["Nickname"])
        } else {
            fmt.Println("No nickname")
        }

        return a, nil
    }
    return nil, sql.ErrNoRows
}
```

### Writing Nulls

```go
func UpdateNickname(db *sql.DB, id int64, nickname *string) error {
    if nickname != nil {
        _, err := db.Exec("UPDATE users SET nickname = ? WHERE id = ?", *nickname, id)
        return err
    }
    _, err := db.Exec("UPDATE users SET nickname = NULL WHERE id = ?", id)
    return err
}

// Using atom
func UpdateFromAtom(db *sql.DB, a *atom.Atom) error {
    nickname := a.StringPtrs["Nickname"]
    return UpdateNickname(db, a.Ints["ID"], nickname)
}
```

## Joins and Nested Data

### Flattened Join

```go
type OrderWithCustomer struct {
    OrderID      int64  `db:"order_id"`
    Total        int64  `db:"total"`
    CustomerName string `db:"customer_name"`
}

func GetOrdersWithCustomers(db *sql.DB) ([]*atom.Atom, error) {
    atomizer, _ := atom.Use[OrderWithCustomer]()
    scanner, _ := atom.ScannerFor(atomizer.Spec())

    rows, _ := db.Query(`
        SELECT o.id as order_id, o.total, c.name as customer_name
        FROM orders o
        JOIN customers c ON o.customer_id = c.id
    `)
    defer rows.Close()

    return scanner.ScanAll(rows, rows.Next)
}
```

### Nested Structs

```go
type Order struct {
    ID       int64    `db:"id"`
    Total    int64    `db:"total"`
    Customer Customer // No db tag - nested struct
}

type Customer struct {
    Name  string `db:"customer_name"`
    Email string `db:"customer_email"`
}

// Query flattens to single row
rows, _ := db.Query(`
    SELECT o.id, o.total, c.name as customer_name, c.email as customer_email
    FROM orders o
    JOIN customers c ON o.customer_id = c.id
`)

// Result has nested structure
atom.Ints["ID"]                           // 123
atom.Ints["Total"]                        // 9999
atom.Nested["Customer"].Strings["Name"]   // "Alice"
atom.Nested["Customer"].Strings["Email"]  // "alice@example.com"
```

## Transactions

```go
func TransferBalance(db *sql.DB, fromID, toID, amount int64) error {
    tx, err := db.Begin()
    if err != nil {
        return err
    }
    defer tx.Rollback()

    // Read current balances
    scanner, _ := atom.ScannerFor(atomizer.Spec())

    rows, _ := tx.Query("SELECT id, balance FROM accounts WHERE id IN (?, ?)", fromID, toID)
    atoms, _ := scanner.ScanAll(rows, rows.Next)
    rows.Close()

    // Find accounts
    var from, to *atom.Atom
    for _, a := range atoms {
        if a.Ints["ID"] == fromID {
            from = a
        } else {
            to = a
        }
    }

    // Update balances
    from.Ints["Balance"] -= amount
    to.Ints["Balance"] += amount

    // Write back
    tx.Exec("UPDATE accounts SET balance = ? WHERE id = ?", from.Ints["Balance"], fromID)
    tx.Exec("UPDATE accounts SET balance = ? WHERE id = ?", to.Ints["Balance"], toID)

    return tx.Commit()
}
```

## Error Handling

```go
func SafeQuery(db *sql.DB) ([]*atom.Atom, error) {
    atomizer, err := atom.Use[User]()
    if err != nil {
        return nil, fmt.Errorf("type registration failed: %w", err)
    }

    scanner, ok := atom.ScannerFor(atomizer.Spec())
    if !ok {
        return nil, fmt.Errorf("scanner not available (check db tags)")
    }

    rows, err := db.Query("SELECT id, name, email FROM users")
    if err != nil {
        return nil, fmt.Errorf("query failed: %w", err)
    }
    defer rows.Close()

    atoms, err := scanner.ScanAll(rows, rows.Next)
    if err != nil {
        return nil, fmt.Errorf("scan failed: %w", err)
    }

    return atoms, nil
}
```

## Testing

### Mock ColScanner

```go
type MockRows struct {
    columns []string
    data    [][]any
    index   int
}

func (m *MockRows) Columns() ([]string, error) {
    return m.columns, nil
}

func (m *MockRows) Scan(dest ...any) error {
    row := m.data[m.index]
    for i, d := range dest {
        switch v := d.(type) {
        case *int64:
            *v = row[i].(int64)
        case *string:
            *v = row[i].(string)
        }
    }
    return nil
}

func (m *MockRows) Err() error { return nil }

func (m *MockRows) Next() bool {
    m.index++
    return m.index < len(m.data)
}

func TestScanner(t *testing.T) {
    atomizer, _ := atom.Use[User]()
    scanner, _ := atom.ScannerFor(atomizer.Spec())

    mock := &MockRows{
        columns: []string{"id", "name", "email"},
        data: [][]any{
            {int64(1), "Alice", "alice@example.com"},
            {int64(2), "Bob", "bob@example.com"},
        },
        index: -1,
    }

    atoms, err := scanner.ScanAll(mock, mock.Next)
    if err != nil {
        t.Fatal(err)
    }

    if len(atoms) != 2 {
        t.Errorf("expected 2 atoms, got %d", len(atoms))
    }
    if atoms[0].Strings["Name"] != "Alice" {
        t.Errorf("expected Alice, got %s", atoms[0].Strings["Name"])
    }
}
```
